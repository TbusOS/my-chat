## 1.1 基础结构

```python
import torch
import torch.nn as nn
import torch.nn.functional as F
import math

class MultiHeadAttention(nn.Module):
    """多头注意力机制"""

    def __init__(self, d_model, num_heads, dropout=0.1):
        super().__init__()
        assert d_model % num_heads == 0

        self.d_model = d_model
        self.num_heads = num_heads
        self.d_k = d_model // num_heads

        # Q, K, V 投影矩阵
        self.W_q = nn.Linear(d_model, d_model)
        self.W_k = nn.Linear(d_model, d_model)
        self.W_v = nn.Linear(d_model, d_model)

        # 输出投影
        self.W_o = nn.Linear(d_model, d_model)

        self.dropout = nn.Dropout(dropout)
```

## 1.2 分头实现

```python
    def split_heads(self, x):
        """将张量分割成多个头

        输入: (batch, seq_len, d_model)
        输出: (batch, num_heads, seq_len, d_k)
        """
        batch, seq_len, d_model = x.shape
        x = x.view(batch, seq_len, self.num_heads, self.d_k)
        return x.permute(0, 2, 1, 3)  # (batch, heads, seq, d_k)
```

## 1.3 注意力计算

```python
    def forward(self, Q, K, V, mask=None):
        batch = Q.size(0)

        # 1. 线性投影
        Q = self.split_heads(self.W_q(Q))
        K = self.split_heads(self.W_k(K))
        V = self.split_heads(self.W_v(V))

        # 2. 计算注意力分数
        # (batch, heads, seq_q, d_k) @ (batch, heads, d_k, seq_k)
        # = (batch, heads, seq_q, seq_k)
        scores = torch.matmul(Q, K.transpose(-2, -1)) / math.sqrt(self.d_k)

        # 3. 应用 mask (可选)
        if mask is not None:
            scores = scores.masked_fill(mask == 0, -1e9)

        # 4. Softmax
        attn_weights = F.softmax(scores, dim=-1)
        attn_weights = self.dropout(attn_weights)

        # 5. 加权求和
        # (batch, heads, seq_q, seq_k) @ (batch, heads, seq_v, d_k)
        # = (batch, heads, seq_q, d_k)
        context = torch.matmul(attn_weights, V)

        # 6. 合并多头
        # (batch, heads, seq_q, d_k) -> (batch, seq_q, d_model)
        context = context.permute(0, 2, 1, 3).contiguous()
        context = context.view(batch, -1, self.d_model)

        # 7. 输出投影
        output = self.W_o(context)

        return output, attn_weights
```

## 1.4 完整代码

```python
import torch
import torch.nn as nn
import torch.nn.functional as F
import math

class MultiHeadAttention(nn.Module):
    """多头注意力机制 - 完整实现"""

    def __init__(self, d_model, num_heads, dropout=0.1):
        super().__init__()
        assert d_model % num_heads == 0

        self.d_model = d_model
        self.num_heads = num_heads
        self.d_k = d_model // num_heads
        self.scale = 1.0 / math.sqrt(self.d_k)

        self.W_q = nn.Linear(d_model, d_model)
        self.W_k = nn.Linear(d_model, d_model)
        self.W_v = nn.Linear(d_model, d_model)
        self.W_o = nn.Linear(d_model, d_model)

        self.dropout = nn.Dropout(dropout)

    def split_heads(self, x):
        batch, seq_len, d_model = x.shape
        x = x.view(batch, seq_len, self.num_heads, self.d_k)
        return x.permute(0, 2, 1, 3)

    def forward(self, Q, K, V, mask=None):
        batch = Q.size(0)

        # 投影 + 分头
        Q = self.split_heads(self.W_q(Q))
        K = self.split_heads(self.W_k(K))
        V = self.split_heads(self.W_v(V))

        # 注意力分数
        scores = torch.matmul(Q, K.transpose(-2, -1)) * self.scale

        if mask is not None:
            scores = scores.masked_fill(mask == 0, -1e9)

        # Softmax + Dropout
        attn = F.softmax(scores, dim=-1)
        attn = self.dropout(attn)

        # 加权求和
        context = torch.matmul(attn, V)

        # 合并 + 输出投影
        context = context.permute(0, 2, 1, 3).contiguous()
        context = context.view(batch, -1, self.d_model)

        return self.W_o(context), attn
```
