## 3.1 数据格式

### 3.1.1 Alpaca 格式

```json
[
  {
    "instruction": "生成一个Python函数",
    "input": "计算斐波那契数列",
    "output": "def fibonacci(n):\n    if n <= 1:\n        return n\n    return fibonacci(n-1) + fibonacci(n-2)"
  },
  {
    "instruction": "解释概念",
    "input": "什么是机器学习？",
    "output": "机器学习是人工智能的一个分支..."
  }
]
```

字段说明：
- `instruction`: 任务指令
- `input`: 输入（可选，有些任务没有）
- `output`: 期望输出

### 3.1.2 指令格式变体

```json
{
  "prompt": "用户: 生成一个Python函数\n助手: ",
  "completion": "def fibonacci(n):\n    if n <= 1:\n        return n\n    return fibonacci(n-1) + fibonacci(n-2)"
}
```

### 3.1.3 多轮对话格式

```json
{
  "messages": [
    {"role": "system", "content": "你是一个Python专家"},
    {"role": "user", "content": "如何定义函数？"},
    {"role": "assistant", "content": "在Python中，使用def关键字..."}
```

## 3.3 数据清洗

### 3.3.1 基本清洗

```python
import json

def clean_data(data):
    """清洗数据"""
    cleaned = []

    for item in data:
        # 检查必要字段
        if "instruction" not in item or "output" not in item:
            continue

        # 去除空值
        if not item["instruction"].strip():
            continue
        if not item["output"].strip():
            continue

        # 去除过长/过短
        if len(item["output"]) < 10:
            continue
        if len(item["output"]) > 2000:
            continue

        cleaned.append(item)

    return cleaned

# 使用
with open("data.json", "r", encoding="utf-8") as f:
    data = json.load(f)

cleaned = clean_data(data)
print(f"清洗后: {len(cleaned)} 条")
```

```python
def filter_quality(data, min_score=0.5):
```

## 3.5 数据集创建实战

### 3.5.1 创建任务数据集

```python
import json
import random

# 定义任务模板
TEMPLATES = [
    {
        "instruction": "用Python实现{}",
        "input": "{}",
        "output_template": "def {}():\n    pass"
    },
    {
        "instruction": "解释什么是{}",
        "input": "",
        "output_template": "{}是..."
    }
]

# 填充数据
def generate_dataset(tasks, count=100):
    """生成训练数据"""
    data = []

    for _ in range(count):
        template = random.choice(tasks)

        # 填充模板
        item = {
            "instruction": template["instruction"].format(*template.get("params", [])),
            "input": template["input"].format(*template.get("params", [])),
            "output": template["output_template"]
        }

        data.append(item)

    return data
```

### 3.5.2 生成 QA 数据

```python
import json

def create_qa_dataset(qa_pairs):
    """从 QA 对创建数据集"""
    data = []

    for q, a in qa_pairs:
        item = {
            "instruction": "请回答以下问题",
            "input": q,
            "output": a
        }
        data.append(item)

    return data

# 使用
qa_pairs = [
    ("Python 是什么?", "Python 是一种高级编程语言..."),
    ("如何定义函数?", "使用 def 关键字..."),
]

dataset = create_qa_dataset(qa_pairs)

with open("data/qa.json", "w", encoding="utf-8") as f:
    json.dump(dataset, f, ensure_ascii=False, indent=2)
```

## 3.6 数据格式转换

### 3.6.1 JSON 转 LitGPT 格式

```python
import json
from litgpt.data import InstructionData

def convert_to_litgpt(input_file, output_file):
    """转换为 LitGPT 格式"""
    with open(input_file, "r", encoding="utf-8") as f:
        data = json.load(f)

    # 转换
    converted = []
    for item in data:
        converted.append({
            "prompt": f"Instruction: {item['instruction']}\n"
                      f"Input: {item.get('input', '')}\n"
                      f"Response:",
            "completion": item["output"]
        })

    # 保存
    with open(output_file, "w", encoding="utf-8") as f:
        json.dump(converted, f, ensure_ascii=False, indent=2)

    print(f"转换完成: {len(converted)} 条")
```
