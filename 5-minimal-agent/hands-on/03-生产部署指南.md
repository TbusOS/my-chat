## 1. 生产级错误处理

### 1.1 重试机制

```python
import time
import requests
from functools import wraps


def retry(max_retries: int = 3, backoff: float = 1.0):
    """指数退避重试装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            last_error = None
            for attempt in range(max_retries):
                try:
                    return func(*args, **kwargs)
                except (requests.Timeout, requests.ConnectionError) as e:
                    last_error = e
                    wait = backoff * (2 ** attempt)
                    print(f"重试 {attempt + 1}/{max_retries}，等待 {wait}s...")
                    time.sleep(wait)
            raise last_error
        return wrapper
    return decorator


class ProductionAgent:
    """生产级 Agent"""

    @retry(max_retries=3, backoff=1.0)
    def _call_llm(self, messages, tools=None):
        resp = requests.post(
            f"{self.base_url}/api/chat",
            json={
                "model": self.model,
                "messages": messages,
                "tools": tools,
                "stream": False
            },
            timeout=60  # 60 秒超时
        )
        resp.raise_for_status()
        return resp.json()
```

### 1.2 工具执行超时

```python
import signal

class ToolTimeout(Exception):
    pass

def execute_with_timeout(func, args, timeout=30):
    """带超时的工具执行"""
    def handler(signum, frame):
        raise ToolTimeout(f"工具执行超过 {timeout} 秒")

    old_handler = signal.signal(signal.SIGALRM, handler)
    signal.alarm(timeout)
    try:
        result = func(**args)
        return str(result)
    except ToolTimeout:
        return "错误: 工具执行超时"
    except Exception as e:
        return f"错误: {e}"
    finally:
        signal.alarm(0)
        signal.signal(signal.SIGALRM, old_handler)
```

## 2. 日志和监控

### 2.1 结构化日志

```python
import logging
import json
from datetime import datetime

# 配置日志
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s [%(levelname)s] %(message)s',
    handlers=[
        logging.FileHandler("agent.log"),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger("agent")


class ObservableAgent:
    """可观测的 Agent"""

    def chat(self, user_input: str) -> str:
        request_id = datetime.now().strftime("%Y%m%d%H%M%S%f")

        logger.info(json.dumps({
            "event": "chat_start",
            "request_id": request_id,
            "input": user_input[:200]  # 截断，避免日志过大
        }))

        start = datetime.now()
        try:
            result = self._process(user_input)
            duration = (datetime.now() - start).total_seconds()

            logger.info(json.dumps({
                "event": "chat_end",
                "request_id": request_id,
                "duration_s": duration,
                "tool_calls_count": self._tool_call_count,
                "output_length": len(result)
            }))
            return result

        except Exception as e:
            logger.error(json.dumps({
                "event": "chat_error",
                "request_id": request_id,
                "error": str(e)
            }))
            raise
```

## 3. 安全最佳实践

### 3.1 输入验证

```python
import re

def validate_input(user_input: str) -> str:
    """验证用户输入"""
    # 限制长度
    if len(user_input) > 4096:
        raise ValueError("输入过长，最多 4096 字符")

    # 过滤注入攻击
    if re.search(r'<script|javascript:|on\w+=', user_input, re.I):
        raise ValueError("检测到潜在注入攻击")

    return user_input.strip()
```

### 3.2 工具白名单

```python
class SecureAgent:
    """安全的 Agent"""

    ALLOWED_TOOLS = {"get_weather", "calculator", "get_time", "search_knowledge"}

    def _execute_tool(self, name: str, args: dict) -> str:
        if name not in self.ALLOWED_TOOLS:
            logger.warning(f"拒绝未授权工具调用: {name}")
            return "错误: 工具未授权"
        return self.tools.execute(name, args)
```

### 3.3 安全计算器（避免 eval 注入）

```python
import ast
import operator

SAFE_OPS = {
    ast.Add: operator.add,
    ast.Sub: operator.sub,
    ast.Mult: operator.mul,
    ast.Div: operator.truediv,
    ast.Pow: operator.pow,
    ast.USub: operator.neg,
}

def safe_eval(expr: str) -> float:
    """安全的数学表达式计算（不使用 eval）"""
    tree = ast.parse(expr, mode='eval')

    def _eval(node):
        if isinstance(node, ast.Expression):
            return _eval(node.body)
        elif isinstance(node, ast.Constant):
            if isinstance(node.value, (int, float)):
                return node.value
            raise ValueError(f"不支持的常量: {node.value}")
        elif isinstance(node, ast.BinOp):
            op = SAFE_OPS.get(type(node.op))
            if not op:
                raise ValueError(f"不支持的运算符: {type(node.op).__name__}")
            return op(_eval(node.left), _eval(node.right))
        elif isinstance(node, ast.UnaryOp):
            op = SAFE_OPS.get(type(node.op))
            if not op:
                raise ValueError(f"不支持的运算符: {type(node.op).__name__}")
            return op(_eval(node.operand))
        raise ValueError(f"不支持的表达式: {type(node).__name__}")

    return _eval(tree)
```

## 5. 常见陷阱

| 陷阱 | 问题 | 解决方案 |
|------|------|----------|
| 无限循环 | LLM 不断调用工具 | 设置 max_tool_rounds |
| 上下文爆炸 | messages 无限增长 | 限制历史长度或做摘要 |
| eval 注入 | 用户通过计算器执行代码 | 用 ast.parse 替代 eval |
| 超时 | LLM 或工具响应太慢 | 设置 timeout + 重试 |
| 内存泄漏 | 长时间运行累积消息 | 定期清理 messages |
| 密钥泄露 | API key 硬编码 | 使用环境变量 |

### 上下文窗口管理

```python
def trim_messages(messages: list, max_messages: int = 20) -> list:
    """裁剪对话历史，保留系统提示和最近的消息"""
    if len(messages) <= max_messages:
        return messages
    # 保留系统提示 + 最近 N 条
    system = [m for m in messages if m["role"] == "system"]
    recent = messages[-(max_messages - len(system)):]
    return system + recent
```
