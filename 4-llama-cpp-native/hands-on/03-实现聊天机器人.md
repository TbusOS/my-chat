## 1.1 基础框架

```cpp
#include <iostream>
#include <string>
#include <vector>
#include "llama.h"

class ChatBot {
private:
    llama_model* model;
    llama_context* ctx;
    std::vector<llama_token> tokens;

public:
    ChatBot(const std::string& model_path) {
        // 加载模型
        llama_model_params mparams = llama_model_default_params();
        model = llama_load_model_from_file(model_path.c_str(), mparams);

        if (!model) {
            throw std::runtime_error("Failed to load model");
        }

        // 创建上下文
        llama_context_params cparams = llama_context_default_params();
        cparams.n_ctx = 2048;
        ctx = llama_new_context_with_model(model, cparams);
    }

    ~ChatBot() {
        if (ctx) llama_free(ctx);
        if (model) llama_free_model(model);
    }

    void add_message(const std::string& role, const std::string& content) {
        // 构建 prompt
        std::string prompt = role + ": " + content + "\n";

        // 分词并添加
        auto new_tokens = llama_tokenize(ctx, prompt.c_str(), true);
        tokens.insert(tokens.end(), new_tokens.begin(), new_tokens.end());
    }

    std::string generate_response(int max_tokens = 256) {
        // 生成
        for (int i = 0; i < max_tokens; i++) {
            llama_decode(ctx, tokens.data(), tokens.size());

            // 采样
            auto logits = llama_get_logits_ith(ctx, 0);
            auto n_vocab = llama_n_vocab(model);

            llama_token token = 0;
            float max_logit = logits[0];
            for (int j = 1; j < n_vocab; j++) {
                if (logits[j] > max_logit) {
                    max_logit = logits[j];
                    token = j;
                }
            }

            // 结束
            if (token == llama_token_eos(ctx)) {
                break;
            }

            // 输出
            std::string piece = llama_token_to_piece(ctx, token);
            std::cout << piece << std::flush;

            // 添加
            tokens.push_back(token);
        }

        return "";
    }
};
```

## 1.2 完整聊天机器人

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <sstream>
#include "llama.h"

class ChatBot {
private:
    llama_model* model = nullptr;
    llama_context* ctx = nullptr;
    std::vector<llama_token> tokens;

public:
    bool init(const std::string& model_path) {
        llama_model_params mparams = llama_model_default_params();
        mparams.n_gpu_layers = 0;

        model = llama_load_model_from_file(model_path.c_str(), mparams);
        if (!model) return false;

        llama_context_params cparams = llama_context_default_params();
        cparams.n_ctx = 2048;
        cparams.n_threads = 4;

        ctx = llama_new_context_with_model(model, cparams);
        return ctx != nullptr;
    }

    std::string chat(const std::string& user_input, int max_tokens = 256) {
        // 构建 prompt
        std::ostringstream prompt;
        prompt << "Instruction: You are a helpful assistant.\n";
        prompt << "Input: " << user_input << "\n";
        prompt << "Output:";

        auto prompt_tokens = llama_tokenize(ctx, prompt.str().c_str(), true);
        tokens.insert(tokens.end(), prompt_tokens.begin(), prompt_tokens.end());

        std::string result;

        for (int i = 0; i < max_tokens; i++) {
            llama_decode(ctx, tokens.data(), tokens.size());
            tokens.clear();

            auto logits = llama_get_logits_ith(ctx, 0);
            auto n_vocab = llama_n_vocab(model);

            // Greedy 采样
            llama_token token = 0;
            float max_logit = logits[0];
            for (int j = 1; j < n_vocab; j++) {
                if (logits[j] > max_logit) {
                    max_logit = logits[j];
                    token = j;
                }
            }

            if (token == llama_token_eos(ctx)) {
                break;
            }

            std::string piece = llama_token_to_piece(ctx, token);
            result += piece;

            tokens.push_back(token);
        }

        return result;
    }

    ~ChatBot() {
        if (ctx) llama_free(ctx);
        if (model) llama_free_model(model);
    }
};

int main(int argc, char** argv) {
    if (argc < 2) {
        std::cout << "Usage: " << argv[0] << " <model.gguf>\n";
        return 1;
    }

    ChatBot bot;

    std::cout << "Loading model...\n";
    if (!bot.init(argv[1])) {
        std::cerr << "Failed to initialize chatbot\n";
        return 1;
    }

    std::cout << "ChatBot ready! Type 'quit' to exit.\n\n";

    while (true) {
        std::cout << "You: ";
        std::string input;
        std::getline(std::cin, input);

        if (input == "quit" || input == "exit") {
            break;
        }

        if (input.empty()) continue;

        std::cout << "Bot: ";
        std::string response = bot.chat(input);
        std::cout << "\n\n";
    }

    return 0;
}
```
