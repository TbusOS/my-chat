## 1.4 C++ API

### 1.4.1 完整代码

```cpp
#include <iostream>
#include "llama.h"

int main() {
    // 1. 加载模型
    struct llama_model_params mparams = llama_model_default_params();
    mparams.n_gpu_layers = 0;  // CPU 推理

    struct llama_model* model = llama_load_model_from_file(
        "models/llama-2-7b-chat.Q4_K_M.gguf",
        mparams
    );

    if (!model) {
        std::cerr << "Failed to load model\n";
        return 1;
    }

    // 2. 创建上下文
    struct llama_context_params cparams = llama_context_default_params();
    cparams.n_ctx = 512;

    struct llama_context* ctx = llama_new_context_with_model(model, cparams);

    // 3. 准备 prompt
    const char* prompt = "Write a hello world in Python:";

    // 4. 分词
    std::vector<llama_token> tokens = llama_tokenize(ctx, prompt, true);

    // 5. 推理
    for (int i = 0; i < 256; i++) {
        llama_decode(ctx, tokens.data(), tokens.size());
        tokens.clear();

        // 采样
        auto logits = llama_get_logits_ith(ctx, 0);
        auto n_vocab = llama_n_vocab(model);

        // Greedy 采样
        llama_token token = 0;
        float max_logit = logits[0];
        for (int j = 1; j < n_vocab; j++) {
            if (logits[j] > max_logit) {
                max_logit = logits[j];
                token = j;
            }
        }

        // 结束检查
        if (token == llama_token_eos(ctx)) {
            break;
        }

        // 输出
        std::cout << llama_token_to_piece(ctx, token);

        // 添加到输入
        tokens.push_back(token);
    }

    std::cout << "\n";

    // 6. 清理
    llama_free(ctx);
    llama_free_model(model);

    return 0;
}
```
