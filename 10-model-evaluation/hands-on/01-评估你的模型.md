### 用 Python 脚本评估 Ollama 模型的 Perplexity

如果你只需要计算 Perplexity，可以用更简单的方式：

```python
import requests
import math

def ollama_perplexity(model, text, base_url="http://localhost:11434"):
    """通过 Ollama API 计算近似困惑度"""
    words = text.split()
    total_log_prob = 0.0
    token_count = 0

    for i in range(1, len(words)):
        context = " ".join(words[:i])
        target = words[i]

        response = requests.post(
            f"{base_url}/api/generate",
            json={
                "model": model,
                "prompt": f"{context} {target}",
                "stream": False,
                "options": {"num_predict": 0, "logprobs": True}
            }
        )

        if response.status_code == 200:
            # 注意：Ollama 对 logprobs 的支持取决于版本
            token_count += 1

    return token_count

# 更实用的方案：直接使用 transformers 库（见下文）
```

## 4. 使用 Hugging Face 模型评估

对于更精确的评估，直接使用 Hugging Face 模型：

```python
import torch
from transformers import AutoModelForCausalLM, AutoTokenizer
import math

def evaluate_model(model_name, eval_texts):
    """评估模型在给定文本上的困惑度"""
    tokenizer = AutoTokenizer.from_pretrained(model_name)
    model = AutoModelForCausalLM.from_pretrained(
        model_name,
        torch_dtype=torch.float16,
        device_map="auto"
    )
    model.eval()

    total_loss = 0.0
    total_tokens = 0

    for text in eval_texts:
        inputs = tokenizer(
            text,
            return_tensors="pt",
            truncation=True,
            max_length=2048
        ).to(model.device)

        with torch.no_grad():
            outputs = model(**inputs, labels=inputs["input_ids"])

        total_loss += outputs.loss.item() * inputs["input_ids"].size(1)
        total_tokens += inputs["input_ids"].size(1)

    avg_loss = total_loss / total_tokens
    perplexity = math.exp(avg_loss)

    return {"perplexity": perplexity, "avg_loss": avg_loss}

# 使用示例
eval_texts = [
    "大语言模型的核心是 Transformer 架构，它通过自注意力机制捕获序列中的长距离依赖关系。",
    "Python 是一种广泛使用的编程语言，以其简洁的语法和丰富的库生态著称。",
]

result = evaluate_model("Qwen/Qwen2.5-0.5B", eval_texts)
print(f"Perplexity: {result['perplexity']:.2f}")
```

## 5. 自定义评估任务

当标准基准无法覆盖你的业务场景时，创建自定义评估任务。

### 创建评估数据集

```python
import json

def create_eval_dataset(qa_pairs, output_path):
    """将问答对转换为评估数据集"""
    dataset = [
        {
            "question": qa["question"],
            "choices": qa["choices"],
            "answer": qa["answer"]
        }
        for qa in qa_pairs
    ]

    with open(output_path, "w", encoding="utf-8") as f:
        for item in dataset:
            f.write(json.dumps(item, ensure_ascii=False) + "\n")

    return len(dataset)

# 示例：创建一个中文知识评估集
qa_pairs = [
    {
        "question": "Python 中用于异常处理的关键字是？",
        "choices": ["try/except", "if/else", "for/in", "while/do"],
        "answer": 0
    },
    {
        "question": "Transformer 架构中，自注意力机制的时间复杂度是？",
        "choices": ["O(n)", "O(n log n)", "O(n^2)", "O(n^3)"],
        "answer": 2
    },
]

count = create_eval_dataset(qa_pairs, "my_eval.jsonl")
print(f"创建了 {count} 道评估题")
```

### 配置自定义任务

创建一个 YAML 配置文件让 lm-evaluation-harness 识别你的任务：

```yaml
# my_task.yaml
task: my_chinese_eval
dataset_path: json
dataset_kwargs:
  data_files: my_eval.jsonl
output_type: multiple_choice
doc_to_text: "问题：{{question}}\nA. {{choices[0]}}\nB. {{choices[1]}}\nC. {{choices[2]}}\nD. {{choices[3]}}\n答案："
doc_to_target: answer
metric_list:
  - metric: acc
    aggregation: mean
    higher_is_better: true
```

```bash
# 运行自定义任务
lm_eval --model hf \
    --model_args pretrained=Qwen/Qwen2.5-0.5B \
    --tasks my_task \
    --include_path ./ \
    --output_path ./results/custom
```

### 多模型对比脚本

```python
import json
from pathlib import Path

def compare_results(result_dirs):
    """对比多个模型的评估结果"""
    comparison = {}

    for result_dir in result_dirs:
        result_path = Path(result_dir)
        results_file = list(result_path.glob("**/results.json"))

        if not results_file:
            continue

        with open(results_file[0], "r") as f:
            data = json.load(f)

        model_name = result_path.name
        comparison[model_name] = {}

        results = data.get("results", {})
        for task_name, metrics in results.items():
            comparison[model_name][task_name] = {
                k: round(v, 4)
                for k, v in metrics.items()
                if isinstance(v, (int, float))
            }

    return comparison

# 对比结果
results = compare_results([
    "./results/qwen2.5-0.5b",
    "./results/llama3.2-3b",
])

for model, tasks in results.items():
    print(f"\n{'='*40}")
    print(f"Model: {model}")
    for task, metrics in tasks.items():
        print(f"  {task}:")
        for metric, value in metrics.items():
            print(f"    {metric}: {value}")
```

## 常见问题

### Q: 评估太慢怎么办？

```bash
# 增大 batch_size（需要更多显存）
lm_eval --model hf \
    --model_args pretrained=model_name \
    --tasks mmlu \
    --batch_size auto:4

# 只评估部分样本
lm_eval --model hf \
    --model_args pretrained=model_name \
    --tasks mmlu \
    --limit 100
```

### Q: 显存不够怎么办？

```bash
# 使用量化模型
lm_eval --model hf \
    --model_args pretrained=model_name,load_in_4bit=True \
    --tasks mmlu

# 减小 batch_size
lm_eval --model hf \
    --model_args pretrained=model_name \
    --tasks mmlu \
    --batch_size 1
```

### Q: 如何确保评估结果可复现？

```bash
# 固定随机种子
lm_eval --model hf \
    --model_args pretrained=model_name \
    --tasks mmlu \
    --seed 42 \
    --output_path ./results/reproducible
```
